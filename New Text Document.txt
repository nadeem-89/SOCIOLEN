const stages = ['applying theories...','cross-referencing...','generating insights...','formulating questions...','almost done...'];
let si=0, stInt;

async function runAnalysis() {
  const topic = document.getElementById('analyzeInput').value.trim();
  if(!topic){alert('Please enter a social topic.');return;}
  const theories = Array.from(document.querySelectorAll('.theory-chip.active[data-theory]')).map(c=>c.dataset.theory);
  if(!theories.length){alert('Select at least one theory.');return;}

  document.getElementById('analyzeBtn').disabled=true;
  document.getElementById('loadingOverlay').classList.add('active');
  document.getElementById('resultContainer').classList.remove('active');

  si=0; stInt=setInterval(()=>{document.getElementById('loadingStage').textContent=stages[si++%stages.length];},1500);

  const isUrdu = selectedResponseLang==='Urdu';
  const langInstr = isUrdu ? 'Respond ENTIRELY in Urdu script (not Roman Urdu). All JSON values must be in Urdu.' : selectedResponseLang.includes('Both') ? 'Provide English, then add Urdu translation after \\n\\n---\\n\\n' : 'Respond in English.';

  const prompt = `You are an expert sociologist. Analyze this topic.
TOPIC: "${topic}"
THEORIES: ${theories.join(', ')}
LANGUAGE: ${langInstr}

Respond ONLY in valid JSON (no markdown, no backticks, no preamble):
{"overview":"2-3 paragraph academic overview","theories":"Apply each theory to this topic. Format: THEORY NAME: analysis (one per line)","ai_methods":"3-4 specific AI/computational methods to study this (NLP, sentiment analysis, big data, ML, social network analysis, predictive modeling)","research_questions":"5 research questions + 2 methodology suggestions","policy":"3-4 concrete policy recommendations","interview_statement":"One impressive sentence for a Sociology Lecturer interview connecting this topic to AI and sociological theory"}`;

  try {
    const res = await fetch("https://api.anthropic.com/v1/messages",{method:"POST",headers:{"Content-Type":"application/json"},body:JSON.stringify({model:"claude-sonnet-4-20250514",max_tokens:1000,messages:[{role:"user",content:prompt}]})});
    const data = await res.json();
    const raw = data.content.map(i=>i.text||'').join('\n').replace(/```json|```/g,'').trim();
    const p = JSON.parse(raw);
    clearInterval(stInt);
    document.getElementById('loadingOverlay').classList.remove('active');
    document.getElementById('analyzeBtn').disabled=false;

    const u = isUrdu;
    ['overview','theories','ai_methods','research_questions','policy'].forEach((k,i) => {
      const ids = ['r-overview','r-theories','r-ai','r-research','r-policy'];
      const el = document.getElementById(ids[i]);
      el.textContent = p[k];
      el.className = 'result-card-body'+(u?' urdu':'');
    });
    document.getElementById('r-interview').textContent = '"'+p.interview_statement+'"';
    document.getElementById('r-interview').className = 'igc-text'+(u?' urdu':'');
    document.getElementById('resultContainer').classList.add('active');
    document.getElementById('resultContainer').scrollIntoView({behavior:'smooth',block:'start'});
  } catch(err) {
    clearInterval(stInt);
    document.getElementById('loadingOverlay').classList.remove('active');
    document.getElementById('analyzeBtn').disabled=false;
    alert('Error: '+err.message);
  }
}
